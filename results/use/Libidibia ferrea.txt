function toggle_vis() {
	var ele = document.getElementById('synonyms');
	var val = ele.className;
	var eletext = document.getElementById('syn_text');
	var measure = document.querySelector('#measurement');
	if (val == 'synhid') {
		ele.className = "synshown";
		ele.style.height = measure.clientHeight+'px';
		eletext.innerHTML = "<b>- Synonyms</b>"
	} else {
		ele.className = "synhid";
		ele.style.height = "0";
		eletext.innerHTML = "<b>+ Synonyms</b>"
	}
}

].PropagationSeed - it has a hard seedcoat and may benefit from scarification before sowing to speed up germination. This can usually be done by pouring a small amount of nearly boiling water on the seeds (being careful not to cook them!) and then soaking them for 12 - 24 hours in warm water. By this time they should have imbibed moisture and swollen - if they have not, then carefully make a nick in the seedcoat (being careful not to damage the embryo) and soak for a further 12 hours before sowing. Sow the seed in a partially shaded position in a nursery seedbed. A germination rate in excess of 60% can be expected, with the seed sprouting within 7 - 15 days[419TitleBrazilian Trees.
